ZOHO.embeddedApp.on("PageLoad", function(data) { // 1. Use getRecord (singular) instead of getAllRecords ZOHO.CRM.API.getRecord({ Entity: "Deals", RecordID: data.EntityId }) .then(function(response) { const deal = response.data[0]; // 2. Build your table with Deal fields let html = `
FieldValue
Name${deal.Deal_Name}
Amount${deal.Amount || '0'}
Stage${deal.Stage}
`; document.getElementById("leadData").innerHTML = html; }) .catch(error => { console.error(error); document.getElementById("leadData").innerHTML = "Error fetching Deal."; }); }); // 3. MANDATORY: This starts the engine ZOHO.embeddedApp.init();
